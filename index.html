Linked List implementation in Java

December 4, 2016Posted by Navin

 

This is LinkedList implementation in java, we are implementing linked list in java with insertion ,deletion in linked list with first insertion, last insertion, insertion at specific position ,first deletion, last deletion etc

Implementation

 

class Link{

int data1;
Link nextLink;

Link(int d1){
data1=d1;
}

public void printLink(){
System.out.println(data1);
}

}
class LinkList{
private Link first;
private Link last;

public LinkList(){
first=null;
}

public boolean isEmpty(){
return first==null;

}
public void insert(int d1){
Link l=new Link(d1);

if(first==null){
l.nextLink=first;
last=first=l;
}else{
last.nextLink=l;
l.nextLink=null;
last=l;
}

}
public void insertFirst(int d1){
Link l=new Link(d1);
l.nextLink=first;
first=l;
}

public void insertAny(int d1,int pos){
Link l=new Link(d1);
int count=0;
last=first;

if(first==null&&pos==0){
l.nextLink=null;
first=l;
}

if(pos==0){
insertFirst(d1);
}else{
System.out.println(last.data1+" "+count);
while(count!=(pos-1)){
last=last.nextLink;
count++;
}
System.out.println(last.data1+" "+count);
if(last.nextLink==null){
last.nextLink=l;
}else{
l.nextLink=last.nextLink;
last.nextLink=l;
}
}

}

public void deleteFirst(){

first=first.nextLink;

}
public void deleteLast(){
Link temp;
temp=first;

while(temp.nextLink!=last){
temp=temp.nextLink;
}

last=temp;
last.nextLink=null;
}

public void printList(){
Link cList=first;
System.out.println("List :");
while(cList!=null){
cList.printLink();
cList=cList.nextLink;

}
System.out.println("");
}
}
public class LinkListTest {

public static void main(String[] args) {

LinkList ll=new LinkList();

ll.insert(1);
ll.insert(2);
ll.insert(3);

ll.printList();

ll.insertFirst(12);
ll.printList();
ll.deleteLast();
ll.insertAny(11, 3);
ll.printList();

}
}
